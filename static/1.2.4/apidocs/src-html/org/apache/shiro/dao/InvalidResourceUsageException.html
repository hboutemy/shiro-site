019    package org.apache.shiro.dao;
020    
021    /**
022     * Root exception indicating invalid or incorrect usage of a data access resource.  This is thrown
023     * typically when incorrectly using the resource or its API.
024     *
025     * @since 1.2
026     */
027    public class InvalidResourceUsageException extends DataAccessException {
028    
029        /**
030         * Constructs an InvalidResourceUsageException with a message explaining the cause of the exception.
031         *
032         * @param message the message explaining the cause of the exception
033         */
034        public InvalidResourceUsageException(String message) {
035            super(message);
036        }
037    
038        /**
039         * Constructs a InvalidResourceUsageException with a message explaining the cause of the exception.
040         *
041         * @param message the explanation
042         * @param cause   the root cause of the exception, typically an API-specific exception
043         */
044        public InvalidResourceUsageException(String message, Throwable cause) {
045            super(message, cause);
046        }
047    }
