157     * Normalize a relative URI path that may have relative values ("/./",
158     * "/../", and so on ) it it.  <strong>WARNING</strong> - This method is
159     * useful only for normalizing application-generated paths.  It does not
160     * try to perform security checks for malicious input.
161     * Normalize operations were was happily taken from org.apache.catalina.util.RequestUtil in
162     * Tomcat trunk, r939305
163     *
164     * @param path Relative path to be normalized
165     * @return normalized path
166     */

168        return normalize(path, Boolean.getBoolean(ALLOW_BACKSLASH));

171    /**
172     * Normalize a relative URI path that may have relative values ("/./",
173     * "/../", and so on ) it it.  <strong>WARNING</strong> - This method is
174     * useful only for normalizing application-generated paths.  It does not
175     * try to perform security checks for malicious input.
176     * Normalize operations were was happily taken from org.apache.catalina.util.RequestUtil in
177     * Tomcat trunk, r939305
178     *
179     * @param path             Relative path to be normalized
180     * @param replaceBackSlash Should '\\' be replaced with '/'
181     * @return normalized path
182     */
183    private static String normalize(String path, boolean replaceBackSlash) {
184
185        if (path == null)
186            return null;
187
188        // Create a place for the normalized path
189        String normalized = path;
190
191        if (replaceBackSlash && normalized.indexOf('\\') >= 0)
192            normalized = normalized.replace('\\', '/');
193
194        if (normalized.equals("/."))
195            return "/";
196
197        // Add a leading "/" if necessary
198        if (!normalized.startsWith("/"))
199            normalized = "/" + normalized;
200
201        // Resolve occurrences of "//" in the normalized path
202        while (true) {
203            int index = normalized.indexOf("//");
204            if (index < 0)
205                break;
206            normalized = normalized.substring(0, index) +
207                    normalized.substring(index + 1);
208        }
209
210        // Resolve occurrences of "/./" in the normalized path
211        while (true) {
212            int index = normalized.indexOf("/./");
213            if (index < 0)
214                break;
215            normalized = normalized.substring(0, index) +
216                    normalized.substring(index + 2);
217        }
218
219        // Resolve occurrences of "/../" in the normalized path
220        while (true) {
221            int index = normalized.indexOf("/../");
222            if (index < 0)
223                break;
224            if (index == 0)
225                return (null);  // Trying to go outside our context
226            int index2 = normalized.lastIndexOf('/', index - 1);
227            normalized = normalized.substring(0, index2) +
228                    normalized.substring(index + 3);
229        }
230
231        // Return the normalized path that we have completed
232        return (normalized);
233
234    }

245        uri = decodeRequestString(request, uri);

249    private static String removeSemicolon(String uri) {
250        int semicolonIndex = uri.indexOf(';');
251        return (semicolonIndex != -1 ? uri.substring(0, semicolonIndex) : uri);
252    }

254    /**
255     * Return the context path for the given request, detecting an include request
256     * URL if called within a RequestDispatcher include.
257     * <p>As the value returned by <code>request.getContextPath()</code> is <i>not</i>
258     * decoded by the servlet container, this method will decode it.
259     *
260     * @param request current HTTP request
261     * @return the context path
262     */
263    public static String getContextPath(HttpServletRequest request) {
264        String contextPath = (String) request.getAttribute(INCLUDE_CONTEXT_PATH_ATTRIBUTE);
265        if (contextPath == null) {
266            contextPath = request.getContextPath();
267        }
268        contextPath = normalize(decodeRequestString(request, contextPath));
269        if ("/".equals(contextPath)) {
270            // the normalize method will return a "/" and includes on Jetty, will also be a "/".
271            contextPath = "";
272        }
273        return contextPath;
274    }

315    /**
316     * Find the Shiro {@link WebEnvironment} for this web application.
317     *
318     * @param sc       ServletContext to find the web application context for
319     * @param attrName the name of the ServletContext attribute to look for
320     * @return the desired WebEnvironment for this web app, or <code>null</code> if none
321     * @since 1.2
322     */
323    public static WebEnvironment getWebEnvironment(ServletContext sc, String attrName) {
324        if (sc == null) {
325            throw new IllegalArgumentException("ServletContext argument must not be null.");
326        }
327        Object attr = sc.getAttribute(attrName);
328        if (attr == null) {
329            return null;
330        }
331        if (attr instanceof RuntimeException) {
332            throw (RuntimeException) attr;
333        }
334        if (attr instanceof Error) {
335            throw (Error) attr;
336        }
337        if (attr instanceof Exception) {
338            throw new IllegalStateException((Exception) attr);
339        }
340        if (!(attr instanceof WebEnvironment)) {
341            throw new IllegalStateException("Context attribute is not of type WebEnvironment: " + attr);
342        }
343        return (WebEnvironment) attr;
344    }

348     * Decode the given source string with a URLDecoder. The encoding will be taken
349     * from the request, falling back to the default "ISO-8859-1".
350     * <p>The default implementation uses <code>URLDecoder.decode(input, enc)</code>.
351     *
352     * @param request current HTTP request
353     * @param source  the String to decode
354     * @return the decoded String
355     * @see #DEFAULT_CHARACTER_ENCODING
356     * @see javax.servlet.ServletRequest#getCharacterEncoding
357     * @see java.net.URLDecoder#decode(String, String)
358     * @see java.net.URLDecoder#decode(String)
359     */
360    @SuppressWarnings({"deprecation"})
361    public static String decodeRequestString(HttpServletRequest request, String source) {
362        String enc = determineEncoding(request);
363        try {
364            return URLDecoder.decode(source, enc);
365        } catch (UnsupportedEncodingException ex) {
366            if (log.isWarnEnabled()) {
367                log.warn("Could not decode request string [" + Encode.forHtml(source) + "] with encoding '" + Encode.forHtml(enc) +
368                        "': falling back to platform default encoding; exception message: " + ex.getMessage());
369            }
370            return URLDecoder.decode(source);
371        }
372    }

588    /**
589     * <p>Checks to see if a request param is considered true using a loose matching strategy for
590     * general values that indicate that something is true or enabled, etc.</p>
591     * <p/>
592     * <p>Values that are considered "true" include (case-insensitive): true, t, 1, enabled, y, yes, on.</p>
593     *
594     * @param request   the servlet request
595     * @param paramName @return true if the param value is considered true or false if it isn't.
596     * @return true if the given parameter is considered "true" - false otherwise.
597     */
598    public static boolean isTrue(ServletRequest request, String paramName) {
599        String value = getCleanParam(request, paramName);
600        return value != null &&
601                (value.equalsIgnoreCase("true") ||
602                        value.equalsIgnoreCase("t") ||
603                        value.equalsIgnoreCase("1") ||
604                        value.equalsIgnoreCase("enabled") ||
605                        value.equalsIgnoreCase("y") ||
606                        value.equalsIgnoreCase("yes") ||
607                        value.equalsIgnoreCase("on"));
608    }

622    public static void saveRequest(ServletRequest request) {
623        Subject subject = SecurityUtils.getSubject();
624        Session session = subject.getSession();
625        HttpServletRequest httpRequest = toHttp(request);
626        SavedRequest savedRequest = new SavedRequest(httpRequest);
627        session.setAttribute(SAVED_REQUEST_KEY, savedRequest);
628    }

630    public static SavedRequest getAndClearSavedRequest(ServletRequest request) {
631        SavedRequest savedRequest = getSavedRequest(request);
632        if (savedRequest != null) {
633            Subject subject = SecurityUtils.getSubject();
634            Session session = subject.getSession();
635            session.removeAttribute(SAVED_REQUEST_KEY);
636        }
637        return savedRequest;
638    }

640    public static SavedRequest getSavedRequest(ServletRequest request) {
641        SavedRequest savedRequest = null;
642        Subject subject = SecurityUtils.getSubject();
643        Session session = subject.getSession(false);
644        if (session != null) {
645            savedRequest = (SavedRequest) session.getAttribute(SAVED_REQUEST_KEY);
646        }
647        return savedRequest;
648    }

650    /**
651     * Redirects the to the request url from a previously
652     * {@link #saveRequest(javax.servlet.ServletRequest) saved} request, or if there is no saved request, redirects the
653     * end user to the specified {@code fallbackUrl}.  If there is no saved request or fallback url, this method
654     * throws an {@link IllegalStateException}.
655     * <p/>
656     * This method is primarily used to support a common login scenario - if an unauthenticated user accesses a
657     * page that requires authentication, it is expected that request is
658     * {@link #saveRequest(javax.servlet.ServletRequest) saved} first and then redirected to the login page. Then,
659     * after a successful login, this method can be called to redirect them back to their originally requested URL, a
660     * nice usability feature.
661     *
662     * @param request     the incoming request
663     * @param response    the outgoing response
664     * @param fallbackUrl the fallback url to redirect to if there is no saved request available.
665     * @throws IllegalStateException if there is no saved request and the {@code fallbackUrl} is {@code null}.
666     * @throws IOException           if there is an error redirecting
667     * @since 1.0
668     */
669    public static void redirectToSavedRequest(ServletRequest request, ServletResponse response, String fallbackUrl)
670            throws IOException {
671        String successUrl = null;
672        boolean contextRelative = true;
673        SavedRequest savedRequest = WebUtils.getAndClearSavedRequest(request);
674        if (savedRequest != null && savedRequest.getMethod().equalsIgnoreCase(AccessControlFilter.GET_METHOD)) {
675            successUrl = savedRequest.getRequestUrl();
676            contextRelative = false;
677        }
678
679        if (successUrl == null) {
680            successUrl = fallbackUrl;
681        }
682
683        if (successUrl == null) {
684            throw new IllegalStateException("Success URL not available via saved request or via the " +
685                    "successUrlFallback method parameter. One of these must be non-null for " +
686                    "issueSuccessRedirect() to work.");
687        }
688
689        WebUtils.issueRedirect(request, response, successUrl, null, contextRelative);
690    }
