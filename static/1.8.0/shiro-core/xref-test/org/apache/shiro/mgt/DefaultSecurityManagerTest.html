67      @Test
68      public void testDefaultConfig() {
69          Subject subject = SecurityUtils.getSubject();
70  
71          AuthenticationToken token = new UsernamePasswordToken("guest", "guest");
72          subject.login(token);
73          assertTrue(subject.isAuthenticated());
74          assertTrue("guest".equals(subject.getPrincipal()));
75          assertTrue(subject.hasRole("guest"));
76  
77          Session session = subject.getSession();
78          session.setAttribute("key", "value");
79          assertEquals(session.getAttribute("key"), "value");
80  
81          subject.logout();
82  
83          assertNull(subject.getSession(false));
84          assertNull(subject.getPrincipal());
85          assertNull(subject.getPrincipals());

122     public void testSubjectReuseAfterLogout() {
123 
124         Subject subject = SecurityUtils.getSubject();
125 
126         AuthenticationToken token = new UsernamePasswordToken("guest", "guest");
127         subject.login(token);
128         assertTrue(subject.isAuthenticated());
129         assertTrue("guest".equals(subject.getPrincipal()));
130         assertTrue(subject.hasRole("guest"));
131 
132         Session session = subject.getSession();
133         Serializable firstSessionId = session.getId();
134 
135         session.setAttribute("key", "value");
136         assertEquals(session.getAttribute("key"), "value");
137 
138         subject.logout();
139 
140         assertNull(subject.getSession(false));
141         assertNull(subject.getPrincipal());
142         assertNull(subject.getPrincipals());
143 
144         subject.login(new UsernamePasswordToken("lonestarr", "vespa"));
145         assertTrue(subject.isAuthenticated());
146         assertTrue("lonestarr".equals(subject.getPrincipal()));
147         assertTrue(subject.hasRole("goodguy"));
148 
149         assertNotNull(subject.getSession());
150         assertFalse(firstSessionId.equals(subject.getSession().getId()));
151 
152         subject.logout();
153 
154         assertNull(subject.getSession(false));
155         assertNull(subject.getPrincipal());
156         assertNull(subject.getPrincipals());
157 
158     }

160     /**
161      * Test ensures that a {@link Subject#login(AuthenticationToken)} first uses
162      * the {@link SecurityManager} passed to its {@link Subject.Builder}
163      * (if one was) instead of the one found in either the {@link ThreadContext}
164      * or statically in {@link SecurityUtils}, either of which may not exist.
165      * <a href="https://issues.apache.org/jira/browse/SHIRO-457">SHIRO-457</a>
166      */
167     @Test
168     public void testNewSubjectWithoutThreadSecurityManager() {
169         // Ensure no fallback sm exists in thread context or statically
170         SecurityUtils.setSecurityManager(null);
171         try {
172             SecurityUtils.getSecurityManager();
173         } catch (UnavailableSecurityManagerException e) {
174             assertTrue(e.getMessage().startsWith("No SecurityManager accessible"));
175         }
176 
177         // Specify sm to use and build subject with
178         DelegatingSubject subject =
179             (DelegatingSubject)(new Subject.Builder(sm)).buildSubject();
180 
181         // Login and verify specified sm is used and no error thrown
182         AuthenticationToken token = new UsernamePasswordToken("guest", "guest");
183         subject.login(token);
184         assertEquals(sm, subject.getSecurityManager());
185     }
