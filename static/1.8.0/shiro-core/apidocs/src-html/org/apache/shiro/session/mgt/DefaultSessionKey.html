016package org.apache.shiro.session.mgt;
017
018import java.io.Serializable;
019
020/**
021 * Default implementation of the {@link SessionKey} interface, which allows setting and retrieval of a concrete
022 * {@link #getSessionId() sessionId} that the {@code SessionManager} implementation can use to look up a
023 * {@code Session} instance.
024 *
025 * @since 1.0
026 */
027public class DefaultSessionKey implements SessionKey, Serializable {
028
029    private Serializable sessionId;
030
031    public DefaultSessionKey() {
032    }
033
034    public DefaultSessionKey(Serializable sessionId) {
035        this.sessionId = sessionId;
036    }
037
038    public void setSessionId(Serializable sessionId) {
039        this.sessionId = sessionId;
040    }
041
042    public Serializable getSessionId() {
043        return this.sessionId;
044    }
045}
