019package org.apache.shiro.authc;
020
021/**
022 * A special kind of <tt>DisabledAccountException</tt>, this exception is thrown when attempting
023 * to authenticate and the corresponding account has been disabled explicitly due to being locked.
024 *
025 * <p>For example, an account can be locked if an administrator explicitly locks an account or
026 * perhaps an account can be locked automatically by the system if too many unsuccessful
027 * authentication attempts take place during a specific period of time (perhaps indicating a
028 * hacking attempt).
029 *
030 * @since 0.1
031 */
032public class LockedAccountException extends DisabledAccountException {
033
034    /**
035     * Creates a new LockedAccountException.
036     */
037    public LockedAccountException() {
038        super();
039    }
040
041    /**
042     * Constructs a new LockedAccountException.
043     *
044     * @param message the reason for the exception
045     */
046    public LockedAccountException(String message) {
047        super(message);
048    }
049
050    /**
051     * Constructs a new LockedAccountException.
052     *
053     * @param cause the underlying Throwable that caused this exception to be thrown.
054     */
055    public LockedAccountException(Throwable cause) {
056        super(cause);
057    }
058
059    /**
060     * Constructs a new LockedAccountException.
061     *
062     * @param message the reason for the exception
063     * @param cause   the underlying Throwable that caused this exception to be thrown.
064     */
065    public LockedAccountException(String message, Throwable cause) {
066        super(message, cause);
067    }
068
069}
