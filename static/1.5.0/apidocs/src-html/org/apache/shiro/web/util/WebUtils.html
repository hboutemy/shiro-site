096    /**
097     * Return the path within the web application for the given request.
098     * Detects include request URL if called within a RequestDispatcher include.
099     * <p/>
100     * For example, for a request to URL
101     * <p/>
102     * <code>http://www.somehost.com/myapp/my/url.jsp</code>,
103     * <p/>
104     * for an application deployed to <code>/mayapp</code> (the application's context path), this method would return
105     * <p/>
106     * <code>/my/url.jsp</code>.
107     *
108     * @param request current HTTP request
109     * @return the path within the web application
110     */
111    public static String getPathWithinApplication(HttpServletRequest request) {
112        String contextPath = getContextPath(request);
113        String requestUri = getRequestUri(request);
114        if (StringUtils.startsWithIgnoreCase(requestUri, contextPath)) {
115            // Normal case: URI contains context path.
116            String path = requestUri.substring(contextPath.length());
117            return (StringUtils.hasText(path) ? path : "/");
118        } else {
119            // Special case: rather unusual.
120            return requestUri;
121        }
122    }

124    /**
125     * Return the request URI for the given request, detecting an include request
126     * URL if called within a RequestDispatcher include.
127     * <p>As the value returned by <code>request.getRequestURI()</code> is <i>not</i>
128     * decoded by the servlet container, this method will decode it.
129     * <p>The URI that the web container resolves <i>should</i> be correct, but some
130     * containers like JBoss/Jetty incorrectly include ";" strings like ";jsessionid"
131     * in the URI. This method cuts off such incorrect appendices.
132     *
133     * @param request current HTTP request
134     * @return the request URI
135     */
136    public static String getRequestUri(HttpServletRequest request) {
137        String uri = (String) request.getAttribute(INCLUDE_REQUEST_URI_ATTRIBUTE);
138        if (uri == null) {
139            uri = request.getRequestURI();
140        }
141        return normalize(decodeAndCleanUriString(request, uri));
142    }

146     * "/../", and so on ) it it.  <strong>WARNING</strong> - This method is
147     * useful only for normalizing application-generated paths.  It does not
148     * try to perform security checks for malicious input.
149     * Normalize operations were was happily taken from org.apache.catalina.util.RequestUtil in
150     * Tomcat trunk, r939305
151     *

153     * @return normalized path
154     */

156        return normalize(path, true);

159    /**
160     * Normalize a relative URI path that may have relative values ("/./",
161     * "/../", and so on ) it it.  <strong>WARNING</strong> - This method is
162     * useful only for normalizing application-generated paths.  It does not
163     * try to perform security checks for malicious input.
164     * Normalize operations were was happily taken from org.apache.catalina.util.RequestUtil in
165     * Tomcat trunk, r939305
166     *
167     * @param path             Relative path to be normalized
168     * @param replaceBackSlash Should '\\' be replaced with '/'
169     * @return normalized path
170     */
171    private static String normalize(String path, boolean replaceBackSlash) {
172
173        if (path == null)
174            return null;
175
176        // Create a place for the normalized path
177        String normalized = path;
178
179        if (replaceBackSlash && normalized.indexOf('\\') >= 0)
180            normalized = normalized.replace('\\', '/');
181
182        if (normalized.equals("/."))
183            return "/";
184
185        // Add a leading "/" if necessary
186        if (!normalized.startsWith("/"))
187            normalized = "/" + normalized;
188
189        // Resolve occurrences of "//" in the normalized path
190        while (true) {
191            int index = normalized.indexOf("//");
192            if (index < 0)
193                break;
194            normalized = normalized.substring(0, index) +
195                    normalized.substring(index + 1);
196        }
197
198        // Resolve occurrences of "/./" in the normalized path
199        while (true) {
200            int index = normalized.indexOf("/./");
201            if (index < 0)
202                break;
203            normalized = normalized.substring(0, index) +
204                    normalized.substring(index + 2);
205        }
206
207        // Resolve occurrences of "/../" in the normalized path
208        while (true) {
209            int index = normalized.indexOf("/../");
210            if (index < 0)
211                break;
212            if (index == 0)
213                return (null);  // Trying to go outside our context
214            int index2 = normalized.lastIndexOf('/', index - 1);
215            normalized = normalized.substring(0, index2) +
216                    normalized.substring(index + 3);
217        }
218
219        // Return the normalized path that we have completed
220        return (normalized);
221
222    }

225    /**
226     * Decode the supplied URI string and strips any extraneous portion after a ';'.
227     *
228     * @param request the incoming HttpServletRequest
229     * @param uri     the application's URI string
230     * @return the supplied URI string stripped of any extraneous portion after a ';'.
231     */
232    private static String decodeAndCleanUriString(HttpServletRequest request, String uri) {
233        uri = decodeRequestString(request, uri);
234        int semicolonIndex = uri.indexOf(';');
235        return (semicolonIndex != -1 ? uri.substring(0, semicolonIndex) : uri);
236    }

238    /**
239     * Return the context path for the given request, detecting an include request
240     * URL if called within a RequestDispatcher include.
241     * <p>As the value returned by <code>request.getContextPath()</code> is <i>not</i>
242     * decoded by the servlet container, this method will decode it.
243     *
244     * @param request current HTTP request
245     * @return the context path
246     */
247    public static String getContextPath(HttpServletRequest request) {
248        String contextPath = (String) request.getAttribute(INCLUDE_CONTEXT_PATH_ATTRIBUTE);
249        if (contextPath == null) {
250            contextPath = request.getContextPath();
251        }
252        contextPath = normalize(decodeRequestString(request, contextPath));
253        if ("/".equals(contextPath)) {
254            // the normalize method will return a "/" and includes on Jetty, will also be a "/".
255            contextPath = "";
256        }
257        return contextPath;
258    }

299    /**
300     * Find the Shiro {@link WebEnvironment} for this web application.
301     *
302     * @param sc       ServletContext to find the web application context for
303     * @param attrName the name of the ServletContext attribute to look for
304     * @return the desired WebEnvironment for this web app, or <code>null</code> if none
305     * @since 1.2
306     */
307    public static WebEnvironment getWebEnvironment(ServletContext sc, String attrName) {
308        if (sc == null) {
309            throw new IllegalArgumentException("ServletContext argument must not be null.");
310        }
311        Object attr = sc.getAttribute(attrName);
312        if (attr == null) {
313            return null;
314        }
315        if (attr instanceof RuntimeException) {
316            throw (RuntimeException) attr;
317        }
318        if (attr instanceof Error) {
319            throw (Error) attr;
320        }
321        if (attr instanceof Exception) {
322            throw new IllegalStateException((Exception) attr);
323        }
324        if (!(attr instanceof WebEnvironment)) {
325            throw new IllegalStateException("Context attribute is not of type WebEnvironment: " + attr);
326        }
327        return (WebEnvironment) attr;
328    }

332     * Decode the given source string with a URLDecoder. The encoding will be taken
333     * from the request, falling back to the default "ISO-8859-1".
334     * <p>The default implementation uses <code>URLDecoder.decode(input, enc)</code>.
335     *
336     * @param request current HTTP request
337     * @param source  the String to decode
338     * @return the decoded String
339     * @see #DEFAULT_CHARACTER_ENCODING
340     * @see javax.servlet.ServletRequest#getCharacterEncoding
341     * @see java.net.URLDecoder#decode(String, String)
342     * @see java.net.URLDecoder#decode(String)
343     */
344    @SuppressWarnings({"deprecation"})
345    public static String decodeRequestString(HttpServletRequest request, String source) {
346        String enc = determineEncoding(request);
347        try {
348            return URLDecoder.decode(source, enc);
349        } catch (UnsupportedEncodingException ex) {
350            if (log.isWarnEnabled()) {
351                log.warn("Could not decode request string [" + Encode.forHtml(source) + "] with encoding '" + Encode.forHtml(enc) +
352                        "': falling back to platform default encoding; exception message: " + ex.getMessage());
353            }
354            return URLDecoder.decode(source);
355        }
356    }

360     * Can be overridden in subclasses.
361     * <p>The default implementation checks the request's
362     * {@link ServletRequest#getCharacterEncoding() character encoding}, and if that
363     * <code>null</code>, falls back to the {@link #DEFAULT_CHARACTER_ENCODING}.

572    /**
573     * <p>Checks to see if a request param is considered true using a loose matching strategy for
574     * general values that indicate that something is true or enabled, etc.</p>
575     * <p/>
576     * <p>Values that are considered "true" include (case-insensitive): true, t, 1, enabled, y, yes, on.</p>
577     *
578     * @param request   the servlet request
579     * @param paramName @return true if the param value is considered true or false if it isn't.
580     * @return true if the given parameter is considered "true" - false otherwise.
581     */
582    public static boolean isTrue(ServletRequest request, String paramName) {
583        String value = getCleanParam(request, paramName);
584        return value != null &&
585                (value.equalsIgnoreCase("true") ||
586                        value.equalsIgnoreCase("t") ||
587                        value.equalsIgnoreCase("1") ||
588                        value.equalsIgnoreCase("enabled") ||
589                        value.equalsIgnoreCase("y") ||
590                        value.equalsIgnoreCase("yes") ||
591                        value.equalsIgnoreCase("on"));
592    }

624    public static SavedRequest getSavedRequest(ServletRequest request) {
625        SavedRequest savedRequest = null;
626        Subject subject = SecurityUtils.getSubject();
627        Session session = subject.getSession(false);
628        if (session != null) {
629            savedRequest = (SavedRequest) session.getAttribute(SAVED_REQUEST_KEY);
630        }
631        return savedRequest;
632    }

634    /**
635     * Redirects the to the request url from a previously
636     * {@link #saveRequest(javax.servlet.ServletRequest) saved} request, or if there is no saved request, redirects the
637     * end user to the specified {@code fallbackUrl}.  If there is no saved request or fallback url, this method
638     * throws an {@link IllegalStateException}.
639     * <p/>
640     * This method is primarily used to support a common login scenario - if an unauthenticated user accesses a
641     * page that requires authentication, it is expected that request is
642     * {@link #saveRequest(javax.servlet.ServletRequest) saved} first and then redirected to the login page. Then,
643     * after a successful login, this method can be called to redirect them back to their originally requested URL, a
644     * nice usability feature.
645     *
646     * @param request     the incoming request
647     * @param response    the outgoing response
648     * @param fallbackUrl the fallback url to redirect to if there is no saved request available.
649     * @throws IllegalStateException if there is no saved request and the {@code fallbackUrl} is {@code null}.
650     * @throws IOException           if there is an error redirecting
651     * @since 1.0
652     */
653    public static void redirectToSavedRequest(ServletRequest request, ServletResponse response, String fallbackUrl)
654            throws IOException {
655        String successUrl = null;
656        boolean contextRelative = true;
657        SavedRequest savedRequest = WebUtils.getAndClearSavedRequest(request);
658        if (savedRequest != null && savedRequest.getMethod().equalsIgnoreCase(AccessControlFilter.GET_METHOD)) {
659            successUrl = savedRequest.getRequestUrl();
660            contextRelative = false;
661        }
662
663        if (successUrl == null) {
664            successUrl = fallbackUrl;
665        }
666
667        if (successUrl == null) {
668            throw new IllegalStateException("Success URL not available via saved request or via the " +
669                    "successUrlFallback method parameter. One of these must be non-null for " +
670                    "issueSuccessRedirect() to work.");
671        }
672
673        WebUtils.issueRedirect(request, response, successUrl, null, contextRelative);
674    }
