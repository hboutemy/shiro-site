019package org.apache.shiro.authz.aop;
020
021import org.apache.shiro.aop.AnnotationResolver;
022
023/**
024 * Checks to see if a @{@link org.apache.shiro.authz.annotation.RequiresUser RequiresUser} annotation
025 * is declared, and if so, ensures the calling <code>Subject</code> is <em>either</em>
026 * {@link org.apache.shiro.subject.Subject#isAuthenticated() authenticated} <b><em>or</em></b> remembered via remember
027 * me services before invoking the method.
028 * <p>
029 * This annotation essentially ensures that <code>subject.{@link org.apache.shiro.subject.Subject#getPrincipal() getPrincipal()} != null</code>.
030 *
031 * @since 0.9.0
032 */
033public class UserAnnotationMethodInterceptor extends AuthorizingAnnotationMethodInterceptor {
034
035    /**
036     * Default no-argument constructor that ensures this interceptor looks for
037     *
038     * {@link org.apache.shiro.authz.annotation.RequiresUser RequiresUser} annotations in a method
039     * declaration.
040     */
041    public UserAnnotationMethodInterceptor() {
042        super( new UserAnnotationHandler() );
043    }
044
045    /**
046     *
047     * @param resolver
048     * @since 1.1
049     */
050    public UserAnnotationMethodInterceptor(AnnotationResolver resolver) {
051        super(new UserAnnotationHandler(), resolver);
052    }
053
054}
