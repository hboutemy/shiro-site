019    package org.apache.shiro.web.tags;
020    
021    import javax.servlet.jsp.JspException;
022    import javax.servlet.jsp.tagext.TagSupport;
023    
024    /**
025     * @since 0.1
026     */
027    public abstract class RoleTag extends SecureTag {
028    
029        //TODO - complete JavaDoc
030    
031        private String name = null;
032    
033        public RoleTag() {
034        }
035    
036        public String getName() {
037            return name;
038        }
039    
040        public void setName(String name) {
041            this.name = name;
042        }
043    
044        public int onDoStartTag() throws JspException {
045            boolean show = showTagBody(getName());
046            if (show) {
047                return TagSupport.EVAL_BODY_INCLUDE;
048            } else {
049                return TagSupport.SKIP_BODY;
050            }
051        }
052    
053        protected abstract boolean showTagBody(String roleName);
054    
055    }
