019package org.apache.shiro.spring.web.config;
020
021import org.apache.shiro.mgt.RememberMeManager;
022import org.apache.shiro.mgt.SessionStorageEvaluator;
023import org.apache.shiro.mgt.SessionsSecurityManager;
024import org.apache.shiro.mgt.SubjectFactory;
025import org.apache.shiro.session.mgt.SessionManager;
026import org.apache.shiro.spring.config.AbstractShiroConfiguration;
027import org.apache.shiro.web.mgt.CookieRememberMeManager;
028import org.apache.shiro.web.mgt.DefaultWebSecurityManager;
029import org.apache.shiro.web.mgt.DefaultWebSessionStorageEvaluator;
030import org.apache.shiro.web.mgt.DefaultWebSubjectFactory;
031import org.apache.shiro.web.servlet.Cookie;
032import org.apache.shiro.web.servlet.SimpleCookie;
033import org.apache.shiro.web.session.mgt.DefaultWebSessionManager;
034import org.apache.shiro.web.session.mgt.ServletContainerSessionManager;
035import org.springframework.beans.factory.annotation.Value;
036
037/**
038 * @since 1.4.0
039 */
040public class AbstractShiroWebConfiguration extends AbstractShiroConfiguration {
041
042    @Value("#{ @environment['shiro.sessionManager.sessionIdCookieEnabled'] ?: true }")
043    protected boolean sessionIdCookieEnabled;
044
045    @Value("#{ @environment['shiro.sessionManager.sessionIdUrlRewritingEnabled'] ?: true }")
046    protected boolean sessionIdUrlRewritingEnabled;
047
048    @Value("#{ @environment['shiro.userNativeSessionManager'] ?: false }")
049    protected boolean useNativeSessionManager;
050
051
052    // Session Cookie info
053
054    @Value("#{ @environment['shiro.sessionManager.cookie.name'] ?: T(org.apache.shiro.web.servlet.ShiroHttpSession).DEFAULT_SESSION_ID_NAME }")
055    protected String sessionIdCookieName;
056
057    @Value("#{ @environment['shiro.sessionManager.cookie.maxAge'] ?: T(org.apache.shiro.web.servlet.SimpleCookie).DEFAULT_MAX_AGE }")
058    protected int sessionIdCookieMaxAge;
059
060    @Value("#{ @environment['shiro.sessionManager.cookie.domain'] ?: null }")
061    protected String sessionIdCookieDomain;
062
063    @Value("#{ @environment['shiro.sessionManager.cookie.path'] ?: null }")
064    protected String sessionIdCookiePath;
065
066    @Value("#{ @environment['shiro.sessionManager.cookie.secure'] ?: false }")
067    protected boolean sessionIdCookieSecure;
068
069
070    // RememberMe Cookie info
071
072    @Value("#{ @environment['shiro.rememberMeManager.cookie.name'] ?: T(org.apache.shiro.web.mgt.CookieRememberMeManager).DEFAULT_REMEMBER_ME_COOKIE_NAME }")
073    protected String rememberMeCookieName;
074
075    @Value("#{ @environment['shiro.rememberMeManager.cookie.maxAge'] ?: T(org.apache.shiro.web.servlet.Cookie).ONE_YEAR }")
076    protected int rememberMeCookieMaxAge;
077
078    @Value("#{ @environment['shiro.rememberMeManager.cookie.domain'] ?: null }")
079    protected String rememberMeCookieDomain;
080
081    @Value("#{ @environment['shiro.rememberMeManager.cookie.path'] ?: null }")
082    protected String rememberMeCookiePath;
083
084    @Value("#{ @environment['shiro.rememberMeManager.cookie.secure'] ?: false }")
085    protected boolean rememberMeCookieSecure;
086
087
088    protected SessionManager nativeSessionManager() {
089        DefaultWebSessionManager webSessionManager = new DefaultWebSessionManager();
090        webSessionManager.setSessionIdCookieEnabled(sessionIdCookieEnabled);
091        webSessionManager.setSessionIdUrlRewritingEnabled(sessionIdUrlRewritingEnabled);
092        webSessionManager.setSessionIdCookie(sessionCookieTemplate());
093
094        webSessionManager.setSessionFactory(sessionFactory());
095        webSessionManager.setSessionDAO(sessionDAO());
096        webSessionManager.setDeleteInvalidSessions(sessionManagerDeleteInvalidSessions);
097
098        return webSessionManager;
099    }
100
101    protected Cookie sessionCookieTemplate() {
102        return buildCookie(
103                sessionIdCookieName,
104                sessionIdCookieMaxAge,
105                sessionIdCookiePath,
106                sessionIdCookieDomain,
107                sessionIdCookieSecure);
108    }
109
110    protected Cookie rememberMeCookieTemplate() {
111        return buildCookie(
112                rememberMeCookieName,
113                rememberMeCookieMaxAge,
114                rememberMeCookiePath,
115                rememberMeCookieDomain,
116                rememberMeCookieSecure);
117    }
118
119    protected Cookie buildCookie(String name, int maxAge, String path, String domain, boolean secure) {
120        Cookie cookie = new SimpleCookie(name);
121        cookie.setHttpOnly(true);
122        cookie.setMaxAge(maxAge);
123        cookie.setPath(path);
124        cookie.setDomain(domain);
125        cookie.setSecure(secure);
126
127        return cookie;
128    }
129
130    @Override
131    protected SessionManager sessionManager() {
132        if (useNativeSessionManager) {
133            return nativeSessionManager();
134        }
135        return new ServletContainerSessionManager();
136    }
137
138    protected RememberMeManager rememberMeManager() {
139        CookieRememberMeManager cookieRememberMeManager = new CookieRememberMeManager();
140        cookieRememberMeManager.setCookie(rememberMeCookieTemplate());
141        return cookieRememberMeManager;
142    }
143
144    @Override
145    protected SubjectFactory subjectFactory() {
146        return new DefaultWebSubjectFactory();
147    }
148
149    @Override
150    protected SessionStorageEvaluator sessionStorageEvaluator() {
151        return new DefaultWebSessionStorageEvaluator();
152    }
153
154    protected SessionsSecurityManager createSecurityManager() {
155
156        DefaultWebSecurityManager securityManager = new DefaultWebSecurityManager();
157        securityManager.setSubjectDAO(subjectDAO());
158        securityManager.setSubjectFactory(subjectFactory());
159        securityManager.setRememberMeManager(rememberMeManager());
160
161        return securityManager;
162    }
163
164    protected ShiroFilterChainDefinition shiroFilterChainDefinition() {
165        DefaultShiroFilterChainDefinition chainDefinition = new DefaultShiroFilterChainDefinition();
166        chainDefinition.addPathDefinition("/**", "authc");
167        return chainDefinition;
168    }
169}
