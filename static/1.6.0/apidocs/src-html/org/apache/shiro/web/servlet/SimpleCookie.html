035/**
036 * Default {@link Cookie Cookie} implementation.  'HttpOnly' is supported out of the box, even on
037 * Servlet {@code 2.4} and {@code 2.5} container implementations, using raw header writing logic and not
038 * {@link javax.servlet.http.Cookie javax.servlet.http.Cookie} objects (which only has 'HttpOnly' support in Servlet
039 * {@code 2.6} specifications and above).
040 *
041 * @since 1.0
042 */

101        this.comment = cookie.getComment();
102        this.domain = cookie.getDomain();
103        this.path = cookie.getPath();

107        this.httpOnly = cookie.isHttpOnly();
108        this.sameSite = cookie.getSameSite();
109    }

194    /**
195     * Returns the Cookie's calculated path setting.  If the {@link javax.servlet.http.Cookie#getPath() path} is {@code null}, then the
196     * {@code request}'s {@link javax.servlet.http.HttpServletRequest#getContextPath() context path}
197     * will be returned. If getContextPath() is the empty string or null then the ROOT_PATH constant is returned.
198     *
199     * @param request the incoming HttpServletRequest
200     * @return the path to be used as the path when the cookie is created or removed
201     */
202    private String calculatePath(HttpServletRequest request) {
203        String path = StringUtils.clean(getPath());
204        if (!StringUtils.hasText(path)) {
205            path = StringUtils.clean(request.getContextPath());
206        }
207
208        //fix for http://issues.apache.org/jira/browse/SHIRO-9:
209        if (path == null) {
210            path = ROOT_PATH;
211        }
212        log.trace("calculated path: {}", path);
213        return path;
214    }

216    public void saveTo(HttpServletRequest request, HttpServletResponse response) {
217
218        String name = getName();
219        String value = getValue();
220        String comment = getComment();
221        String domain = getDomain();
222        String path = calculatePath(request);
223        int maxAge = getMaxAge();
224        int version = getVersion();
225        boolean secure = isSecure();
226        boolean httpOnly = isHttpOnly();
227        SameSiteOptions sameSite = getSameSite();
228
229        addCookieHeader(response, name, value, comment, domain, path, maxAge, version, secure, httpOnly, sameSite);
230    }
231
232    private void addCookieHeader(HttpServletResponse response, String name, String value, String comment,
233                                 String domain, String path, int maxAge, int version,
234                                 boolean secure, boolean httpOnly, SameSiteOptions sameSite) {
235
236        String headerValue = buildHeaderValue(name, value, comment, domain, path, maxAge, version, secure, httpOnly, sameSite);
237        response.addHeader(COOKIE_HEADER_NAME, headerValue);
238
239        if (log.isDebugEnabled()) {
240            log.debug("Added HttpServletResponse Cookie [{}]", headerValue);
241        }
242    }

244    /*
245     * This implementation followed the grammar defined here for convenience:
246     * <a href="http://github.com/abarth/http-state/blob/master/notes/2009-11-07-Yui-Naruse.txt">Cookie grammar</a>.
247     *
248     * @return the 'Set-Cookie' header value for this cookie instance.
249     */
250
251    protected String buildHeaderValue(String name, String value, String comment,
252                                      String domain, String path, int maxAge, int version,
253                                      boolean secure, boolean httpOnly) {
254
255        return buildHeaderValue(name, value, comment, domain, path, maxAge, version, secure, httpOnly, getSameSite());
256    }
257
258    protected String buildHeaderValue(String name, String value, String comment,
259                                      String domain, String path, int maxAge, int version,
260                                      boolean secure, boolean httpOnly, SameSiteOptions sameSite) {
261
262        if (!StringUtils.hasText(name)) {
263            throw new IllegalStateException("Cookie name cannot be null/empty.");
264        }
265
266        StringBuilder sb = new StringBuilder(name).append(NAME_VALUE_DELIMITER);
267
268        if (StringUtils.hasText(value)) {
269            sb.append(value);
270        }
271
272        appendComment(sb, comment);
273        appendDomain(sb, domain);
274        appendPath(sb, path);
275        appendExpires(sb, maxAge);
276        appendVersion(sb, version);
277        appendSecure(sb, secure);
278        appendHttpOnly(sb, httpOnly);
279        appendSameSite(sb, sameSite);
280
281        return sb.toString();
282
283    }
284
285    private void appendComment(StringBuilder sb, String comment) {
286        if (StringUtils.hasText(comment)) {
287            sb.append(ATTRIBUTE_DELIMITER);
288            sb.append(COMMENT_ATTRIBUTE_NAME).append(NAME_VALUE_DELIMITER).append(comment);
289        }
290    }
291
292    private void appendDomain(StringBuilder sb, String domain) {
293        if (StringUtils.hasText(domain)) {
294            sb.append(ATTRIBUTE_DELIMITER);
295            sb.append(DOMAIN_ATTRIBUTE_NAME).append(NAME_VALUE_DELIMITER).append(domain);
296        }
297    }
298
299    private void appendPath(StringBuilder sb, String path) {
300        if (StringUtils.hasText(path)) {
301            sb.append(ATTRIBUTE_DELIMITER);
302            sb.append(PATH_ATTRIBUTE_NAME).append(NAME_VALUE_DELIMITER).append(path);
303        }
304    }
305
306    private void appendExpires(StringBuilder sb, int maxAge) {
307        // if maxAge is negative, cookie should should expire when browser closes
308        // Don't write the maxAge cookie value if it's negative - at least on Firefox it'll cause the 
309        // cookie to be deleted immediately
310        // Write the expires header used by older browsers, but may be unnecessary
311        // and it is not by the spec, see http://www.faqs.org/rfcs/rfc2965.html
312        // TODO consider completely removing the following 
313        if (maxAge >= 0) {
314            sb.append(ATTRIBUTE_DELIMITER);
315            sb.append(MAXAGE_ATTRIBUTE_NAME).append(NAME_VALUE_DELIMITER).append(maxAge);
316            sb.append(ATTRIBUTE_DELIMITER);
317            Date expires;
318            if (maxAge == 0) {
319                //delete the cookie by specifying a time in the past (1 day ago):
320                expires = new Date(System.currentTimeMillis() - DAY_MILLIS);
321            } else {
322                //Value is in seconds.  So take 'now' and add that many seconds, and that's our expiration date:
323                Calendar cal = Calendar.getInstance();
324                cal.add(Calendar.SECOND, maxAge);
325                expires = cal.getTime();
326            }
327            String formatted = toCookieDate(expires);
328            sb.append(EXPIRES_ATTRIBUTE_NAME).append(NAME_VALUE_DELIMITER).append(formatted);
329        }
330    }
331
332    private void appendVersion(StringBuilder sb, int version) {
333        if (version > DEFAULT_VERSION) {
334            sb.append(ATTRIBUTE_DELIMITER);
335            sb.append(VERSION_ATTRIBUTE_NAME).append(NAME_VALUE_DELIMITER).append(version);
336        }
337    }
338
339    private void appendSecure(StringBuilder sb, boolean secure) {
340        if (secure) {
341            sb.append(ATTRIBUTE_DELIMITER);
342            sb.append(SECURE_ATTRIBUTE_NAME); //No value for this attribute
343        }
344    }
345
346    private void appendHttpOnly(StringBuilder sb, boolean httpOnly) {
347        if (httpOnly) {
348            sb.append(ATTRIBUTE_DELIMITER);
349            sb.append(HTTP_ONLY_ATTRIBUTE_NAME); //No value for this attribute
350        }
351    }
352
353    private void appendSameSite(StringBuilder sb, SameSiteOptions sameSite) {
354        if (sameSite != null) {
355            sb.append(ATTRIBUTE_DELIMITER);
356            sb.append(SAME_SITE_ATTRIBUTE_NAME).append(NAME_VALUE_DELIMITER).append(sameSite.toString().toLowerCase(Locale.ENGLISH));
357        }
358    }

360    /**
361     * Check whether the given {@code cookiePath} matches the {@code requestPath}
362     *
363     * @param cookiePath
364     * @param requestPath
365     * @return
366     * @see <a href="https://tools.ietf.org/html/rfc6265#section-5.1.4">RFC 6265, Section 5.1.4 "Paths and Path-Match"</a>
367     */
368    private boolean pathMatches(String cookiePath, String requestPath) {
369        if (!requestPath.startsWith(cookiePath)) {
370            return false;
371        }
372
373        return requestPath.length() == cookiePath.length()
374            || cookiePath.charAt(cookiePath.length() - 1) == '/'
375            || requestPath.charAt(cookiePath.length()) == '/';
376    }

378    /**
379     * Formats a date into a cookie date compatible string (Netscape's specification).
380     *
381     * @param date the date to format
382     * @return an HTTP 1.0/1.1 Cookie compatible date string (GMT-based).
383     */
384    private static String toCookieDate(Date date) {
385        TimeZone tz = TimeZone.getTimeZone(GMT_TIME_ZONE_ID);
386        DateFormat fmt = new SimpleDateFormat(COOKIE_DATE_FORMAT_STRING, Locale.US);
387        fmt.setTimeZone(tz);
388        return fmt.format(date);
389    }

391    public void removeFrom(HttpServletRequest request, HttpServletResponse response) {
392        String name = getName();
393        String value = DELETED_COOKIE_VALUE;
394        String comment = null; //don't need to add extra size to the response - comments are irrelevant for deletions
395        String domain = getDomain();
396        String path = calculatePath(request);
397        int maxAge = 0; //always zero for deletion
398        int version = getVersion();
399        boolean secure = isSecure();
400        boolean httpOnly = false; //no need to add the extra text, plus the value 'deleteMe' is not sensitive at all
401        SameSiteOptions sameSite = null;
402
403        addCookieHeader(response, name, value, comment, domain, path, maxAge, version, secure, httpOnly, sameSite);
404
405        log.trace("Removed '{}' cookie by setting maxAge=0", name);
406    }

408    public String readValue(HttpServletRequest request, HttpServletResponse ignored) {
409        String name = getName();
410        String value = null;
411        javax.servlet.http.Cookie cookie = getCookie(request, name);
412        if (cookie != null) {
413            // Validate that the cookie is used at the correct place.
414            String path = StringUtils.clean(getPath());
415            if (path != null && !pathMatches(path, request.getRequestURI())) {
416                log.warn("Found '{}' cookie at path '{}', but should be only used for '{}'", 
417                                new Object[] { name, Encode.forHtml(request.getRequestURI()), path});
418            } else {
419                value = cookie.getValue();
420                log.debug("Found '{}' cookie value [{}]", name, Encode.forHtml(value));
421            }
422        } else {
423            log.trace("No '{}' cookie value", name);
424        }
425
426        return value;
427    }

429    /**
430     * Returns the cookie with the given name from the request or {@code null} if no cookie
431     * with that name could be found.
432     *
433     * @param request    the current executing http request.
434     * @param cookieName the name of the cookie to find and return.
435     * @return the cookie with the given name from the request or {@code null} if no cookie
436     *         with that name could be found.
437     */
438    private static javax.servlet.http.Cookie getCookie(HttpServletRequest request, String cookieName) {
439        javax.servlet.http.Cookie cookies[] = request.getCookies();
440        if (cookies != null) {
441            for (javax.servlet.http.Cookie cookie : cookies) {
442                if (cookie.getName().equals(cookieName)) {
443                    return cookie;
444                }
445            }
446        }
447        return null;
448    }
