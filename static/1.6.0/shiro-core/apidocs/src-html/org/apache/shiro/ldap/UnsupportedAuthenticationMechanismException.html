019package org.apache.shiro.ldap;
020
021import org.apache.shiro.dao.InvalidResourceUsageException;
022
023/**
024 * Exception thrown when a configured LDAP
025 * <a href="http://download.oracle.com/javase/jndi/tutorial/ldap/security/auth.html">
026 * Authentication Mechanism</a> is unsupported by the target LDAP server. (e.g. DIGEST-MD5, simple, etc)
027 *
028 * @since 1.2
029 */
030public class UnsupportedAuthenticationMechanismException extends InvalidResourceUsageException {
031
032    public UnsupportedAuthenticationMechanismException(String message) {
033        super(message);
034    }
035
036    public UnsupportedAuthenticationMechanismException(String message, Throwable cause) {
037        super(message, cause);
038    }
039}
